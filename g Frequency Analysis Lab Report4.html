Introduction

Fourier series are widely used in electronics to represent periodic signals as sums of sinusoids, making frequency-domain behavior easier to analyze.

They are central to practical systems such as filters and communication links where bandwidth, distortion, and noise must be understood and controlled.

The purpose of this lab is to compare theoretical and measured Fourier spectra of basic waveforms and to study the behavior of an RC low-pass filter.

Frequency Analysis ELA24

Theory

The Fourier series coefficients used later in the results are listed below. Each equation includes a short label and one-sentence interpretation.

  • Eq. (1): Sawtooth harmonics (all harmonics)

    $$A_n = \dfrac{2A}{\pi n},\; n\ge 1$$

    Sawtooth contains all harmonics with amplitudes decreasing approximately as 1/n.

  • Eq. (2): Triangle harmonics (odd only)

    $$A_n = \dfrac{8A}{\pi^2 n^2},\; n=1,3,5,\dots$$

    Triangle contains only odd harmonics with amplitudes decreasing approximately as 1/n^2.

  • Eq. (3): Sine harmonic content

    $$A_1 = A,\; A_{n>1}=0$$

    An ideal sine contains only the fundamental.

  • Eq. (4): RMS and dBV conversion

    $$V_{\mathrm{rms}} = \dfrac{V_{\mathrm{peak}}}{\sqrt{2}},\qquad \mathrm{dBV} = 20\log_{10}\!\left(\dfrac{V_{\mathrm{rms}}}{1\,\mathrm{V}}\right)$$

    These relations convert peak values to RMS and express amplitudes on a dBVrms scale.

  • Eq. (5): RC low-pass transfer function and cutoff

    $$|H(f)| = \dfrac{1}{\sqrt{1 + (f/f_c)^2}},\qquad \angle H(f) = -\arctan(f/f_c),\qquad f_c = \dfrac{1}{2\pi RC}$$

    The RC filter attenuates higher frequencies; above the cutoff, magnitude falls with roughly 20 dB/decade.

Sources for formulas and lab guidance: [1][3]. \nopagebreak[4]

    n = np.asarray(n, int)

# === ELA24: Comparisons in dBVrms (core figures up to Fig. 5) ===
import numpy as np, pandas as pd, matplotlib.pyplot as plt
from IPython.display import HTML, display

plt.rcParams['figure.dpi'] = 120
plt.rcParams['figure.figsize'] = (5.2, 3.2)

F0, VPP = 1000.0, 5.0
A = VPP/2.0; N_HARM = 10
R, C = 8200.0, 10e-9
FC = 1.0/(2*np.pi*R*C)

def vrms_to_dBVrms(vrms, floor=1e-12):
    v = np.maximum(np.asarray(vrms, float), floor); return 20.0*np.log10(v)

def rc_mag(f, fc=FC):
    f = np.asarray(f, float); return 1.0/np.sqrt(1.0 + (f/fc)**2)

def sawtooth_dBVrms(n):
    n = np.asarray(n, int); vpk = 2.0*A/(np.pi*np.maximum(n,1)); return vrms_to_dBVrms(vpk/np.sqrt(2.0))

def triangle_dBVrms(n):
    n = np.asarray(n, int); vrms = np.zeros_like(n, float); odd = (n%2==1)
    vpk_odd = 8.0*A/(np.pi**2 * np.maximum(n[odd],1)**2); vrms[odd] = vpk_odd/np.sqrt(2.0)
    dB = vrms_to_dBVrms(vrms); dB[~odd] = np.nan; return dB

def sine_dBVrms(n):
    n = np.asarray(n, int); vrms = np.zeros_like(n, float); vrms[n==1] = A/np.sqrt(2.0)
    dB = vrms_to_dBVrms(vrms); dB[n!=1] = np.nan; return dB

def theory_dBVrms(F0=F0, N=N_HARM, waveform='saw', apply_rc=False):
    n = np.arange(1, N+1)
    if waveform=='saw': dB = sawtooth_dBVrms(n)
    elif waveform=='tri': dB = triangle_dBVrms(n)
    elif waveform=='sin': dB = sine_dBVrms(n)
    else: raise ValueError
    if apply_rc: dB += vrms_to_dBVrms(rc_mag(n*F0))
    return pd.DataFrame({'n': n, 'f_Hz': n*F0, 'theory_dBVrms': dB})

def read_scope_fft_vrms(path):
    raw = pd.read_csv(path, sep=';', header=None, engine='python', dtype=str)
    freqs, mags = [], []
    for _, row in raw.iterrows():
        nums=[]
        for cell in row.dropna():
            s=cell.strip().replace(',', '.');
            if not s: continue
            try: nums.append(float(s))
            except: pass
        if len(nums)>=2:
            f,v=nums[-2], nums[-1]
            if f>=0 and v>=0: freqs.append(f); mags.append(v)
    spec = pd.DataFrame({'freq': freqs,'Vrms': mags}).sort_values('freq').drop_duplicates('freq').reset_index(drop=True)
    if spec.empty: raise ValueError('No numeric spectrum data parsed')
    spec['dBVrms']=vrms_to_dBVrms(spec['Vrms'])
    return spec

def pick_harmonic_bins(spec_df, F0, N):
    freqs = spec_df['freq'].values; n_all = np.arange(1, N+1); targets = n_all*F0
    ok = (targets>=freqs[0]) & (targets<=freqs[-1])
    n = n_all[ok]; targets = targets[ok]
    idx = np.searchsorted(freqs, targets); idx = np.clip(idx,1,len(freqs)-1)
    chosen = np.where(np.abs(freqs[idx]-targets) < np.abs(freqs[idx-1]-targets), idx, idx-1)
    return pd.DataFrame({'n': n,'f_target_Hz': targets,'f_bin_Hz': freqs[chosen],'meas_dBVrms': spec_df['dBVrms'].values[chosen]})

def caption(t): display(HTML(f"<div class='caption'>{t}</div>"))

def simple_plot(csv_path, waveform, title, apply_rc=False):
    spec = read_scope_fft_vrms(csv_path)
    meas = pick_harmonic_bins(spec, F0, N_HARM)
    theo = theory_dBVrms(waveform=waveform, apply_rc=apply_rc)
    df = meas.merge(theo, on='n', how='left'); df['err_dB']=df['meas_dBVrms']-df['theory_dBVrms']
    plt.figure();
    plt.stem(df['n'], df['theory_dBVrms'], linefmt='C0-', markerfmt='C0o', basefmt=' ', label='Theory')
    plt.stem(df['n']+0.08, df['meas_dBVrms'], linefmt='C1-', markerfmt='C1s', basefmt=' ', label='Measurement')
    plt.xlabel('Harmonic order n'); plt.ylabel('Amplitude [dBVrms]'); plt.title(title); plt.grid(True, alpha=0.35); plt.legend(); plt.tight_layout(); plt.show()
    caption(title)
    return df

Discussion

    vrms[odd] = vpk_odd/np.sqrt(2.0)
    dB = vrms_to_dBVrms(vrms)
    dB[~odd] = np.nan
    return dB

def sine_dBVrms(n, A=A):
    n = np.asarray(n, int)
    vrms = np.zeros_like(n, float)
    vrms[n == 1] = A/np.sqrt(2.0)
    dB = vrms_to_dBVrms(vrms)
    dB[n != 1] = np.nan
    return dB

def theory_dBVrms(F0=F0, N=N_HARM, waveform='saw', apply_rc=False):
    n = np.arange(1, N+1)
    if waveform == 'saw':
        dB = sawtooth_dBVrms(n)
    elif waveform == 'tri':
        dB = triangle_dBVrms(n)
    elif waveform == 'sin':
        dB = sine_dBVrms(n)
    else:
        raise ValueError("waveform must be 'saw' | 'tri' | 'sin'")
    if apply_rc:
        dB += vrms_to_dBVrms(rc_mag(n*F0))  # add 20*log10|H|
    return pd.DataFrame({'n': n, 'f_Hz': n*F0, 'theory_dBVrms': dB})

def read_scope_fft_vrms(csv_path):
    raw = pd.read_csv(csv_path, sep=';', header=None, engine='python', dtype=str)
    freqs, mags = [], []
    for _, row in raw.iterrows():
        nums = []
        for cell in row.dropna():
            s = cell.strip()
            if not s:
                continue
            s = s.replace(',', '.')
            try:
                nums.append(float(s))
            except ValueError:
                pass
        if len(nums) >= 2:
            f, v = nums[-2], nums[-1]
            if f >= 0 and v >= 0:
                freqs.append(f); mags.append(v)
    spec = pd.DataFrame({'freq': freqs, 'Vrms': mags}).sort_values('freq')
    spec = spec.drop_duplicates(subset=['freq']).reset_index(drop=True)
    if spec.empty:
        raise ValueError(f"No numeric data parsed from {csv_path}")
    spec['dBVrms'] = vrms_to_dBVrms(spec['Vrms'].values)
    return spec

def pick_harmonic_bins(spec_df, F0, N_HARM):
    freqs = spec_df['freq'].values
    n_all = np.arange(1, N_HARM+1)
    targets = n_all * F0
    ok = (targets >= freqs[0]) & (targets <= freqs[-1])
    n = n_all[ok]; targets = targets[ok]
    idx = np.searchsorted(freqs, targets)
    idx = np.clip(idx, 1, len(freqs)-1)
    chosen = np.where(np.abs(freqs[idx]-targets) < np.abs(freqs[idx-1]-targets), idx, idx-1)
    return pd.DataFrame({
        'n': n,
        'f_target_Hz': targets,
        'f_bin_Hz': freqs[chosen],
        'meas_dBVrms': spec_df['dBVrms'].values[chosen],
    })

def compare_plot(csv_path, waveform, apply_rc_to_theory=False, title=None, F0=F0, N=N_HARM):
    spec = read_scope_fft_vrms(csv_path)
    meas = pick_harmonic_bins(spec, F0, N)
    theo = theory_dBVrms(F0=F0, N=N, waveform=waveform, apply_rc=apply_rc_to_theory)
    df = meas.merge(theo, on='n', how='left')
    df['err_dB'] = df['meas_dBVrms'] - df['theory_dBVrms']

    plt.figure()
    plt.stem(df['n'],       df['theory_dBVrms'], linefmt='C0-', markerfmt='C0o', basefmt=" ", label='Theory (dBVrms)')
    plt.stem(df['n']+0.08,  df['meas_dBVrms'],   linefmt='C1-', markerfmt='C1s', basefmt=" ", label='Measurement (dBVrms)')
    plt.xlabel('Harmonic order n'); plt.ylabel('Amplitude [dBVrms]')
    ttl = title or f'{Path(csv_path).name}  {waveform.upper()}'
    if apply_rc_to_theory: ttl += ' (after RC, theory  |H|)'
    plt.title(ttl); plt.grid(True, alpha=0.35); plt.legend(); plt.tight_layout(); plt.show()
    return df

def vrms_to_dBVrms(vrms, floor=1e-12):
    v = np.maximum(np.asarray(vrms, float), floor)
    return 20.0*np.log10(v)

# RC low-pass magnitude |H(f)|
def rc_mag(f, fc):
    f = np.asarray(f, float)
    return 1.0/np.sqrt(1.0 + (f/fc)**2)
